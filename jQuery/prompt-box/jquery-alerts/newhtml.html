警告提示框（Alert） 
            $("#alert_button").click( function() {
            jAlert('这是一个可自定义样式的警告对话框', '警告对话框');
        });

确认对话框（Confirm） 
        $("#confirm_button").click( function(){
                jConfirm('你确定这么做吗?', '确认对话框',function(r){
                jAlert('是否确定 ' + r, '确定内容');
            });
        });

带输入的提示框（Prompt） 
            jPrompt('输入字符:', '这里是默认值', '带输入框的提示框', function(r) {
                if( r ) alert('你输入的是 ' + r);
            });
